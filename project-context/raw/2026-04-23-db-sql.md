## Triggers

### Inicio real

```sql
CREATE OR REPLACE TRIGGER ADMIN.TRG_TAREA_INICIO_REAL
BEFORE UPDATE OF ID_ESTADO ON ADMIN.TAREAS
FOR EACH ROW
BEGIN
    IF :NEW.ID_ESTADO = 2
       AND NVL(:OLD.ID_ESTADO, -1) <> 2
       AND :OLD.FECHA_INICIO_REAL IS NULL THEN
        :NEW.FECHA_INICIO_REAL := SYSTIMESTAMP;
    END IF;
END;
/
```

### Fin real

```sql
CREATE OR REPLACE TRIGGER ADMIN.TRG_TAREA_FIN_REAL
BEFORE UPDATE OF ID_ESTADO ON ADMIN.TAREAS
FOR EACH ROW
DECLARE
    V_HORAS NUMBER(10,2);
BEGIN
    IF :NEW.ID_ESTADO = 4
       AND NVL(:OLD.ID_ESTADO, -1) <> 4 THEN

        :NEW.FECHA_FIN_REAL := SYSTIMESTAMP;

        IF :OLD.FECHA_INICIO_REAL IS NOT NULL THEN
            V_HORAS := ROUND(
                (CAST(:NEW.FECHA_FIN_REAL AS DATE) - CAST(:OLD.FECHA_INICIO_REAL AS DATE)) * 24,
                2
            );

            :NEW.HORAS_REALES := V_HORAS;
        END IF;
    END IF;
END;
/
```

## Consultas KPI

### Total de tareas por estado

```sql
SELECT e.NOMBRE_ESTADO,
       COUNT(t.ID_TAREA) AS TOTAL_TAREAS
FROM ADMIN.ESTADOS_TAREA e
LEFT JOIN ADMIN.TAREAS t
    ON t.ID_ESTADO = e.ID_ESTADO
   AND t.ELIMINADA = 0
GROUP BY e.NOMBRE_ESTADO
ORDER BY TOTAL_TAREAS DESC;
```

### Total de tareas por sprint

```sql
SELECT s.NOMBRE AS SPRINT,
       COUNT(t.ID_TAREA) AS TOTAL_TAREAS
FROM ADMIN.SPRINTS s
LEFT JOIN ADMIN.TAREAS t
    ON t.ID_SPRINT = s.ID_SPRINT
   AND t.ELIMINADA = 0
GROUP BY s.NOMBRE
ORDER BY s.ID_SPRINT;
```

### Porcentaje de tareas completadas

```sql
SELECT ROUND(
    (SUM(CASE WHEN e.NOMBRE_ESTADO = 'Completada' THEN 1 ELSE 0 END) * 100.0) /
    NULLIF(COUNT(t.ID_TAREA), 0),
    2
) AS PORCENTAJE_COMPLETADAS
FROM ADMIN.TAREAS t
JOIN ADMIN.ESTADOS_TAREA e
    ON t.ID_ESTADO = e.ID_ESTADO
WHERE t.ELIMINADA = 0;
```

### Tareas completadas por sprint

```sql
SELECT s.NOMBRE AS SPRINT,
       COUNT(t.ID_TAREA) AS TAREAS_COMPLETADAS
FROM ADMIN.SPRINTS s
LEFT JOIN ADMIN.TAREAS t
    ON t.ID_SPRINT = s.ID_SPRINT
   AND t.ELIMINADA = 0
LEFT JOIN ADMIN.ESTADOS_TAREA e
    ON t.ID_ESTADO = e.ID_ESTADO
WHERE e.NOMBRE_ESTADO = 'Completada'
GROUP BY s.NOMBRE
ORDER BY s.ID_SPRINT;
```
